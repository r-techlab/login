// POST endpoint for secure login authentication
function doPost(e) {
  try {
    var requestData = JSON.parse(e.postData.contents);
    var action = requestData.action || "login";
    
    if (action === "login") {
      return handleLogin(requestData, e);
    } else if (action === "validateSession") {
      return handleValidateSession(requestData, e);
    } else if (action === "logout") {
      return handleLogout(requestData, e);
    } else {
      return createJsonResponse({ 
        status: "error", 
        message: "Invalid action" 
      });
    }
  } catch (error) {
    return createJsonResponse({ 
      status: "error", 
      message: "Server error: " + error.toString()
    });
  }
}

// GET endpoint for JSONP requests (handles CORS)
function doGet(e) {
  var action = e.parameter.action || "login";
  
  if (action === "login") {
    return handleLoginGet(e);
  } else if (action === "validateSession") {
    return handleValidateSessionGet(e);
  } else if (action === "logout") {
    return handleLogoutGet(e);
  } else if (action === "getUsers") {
    return handleGetUsersGet(e);
  } else if (action === "getRoles") {
    return handleGetRolesGet(e);
  } else if (action === "createUser") {
    return handleCreateUserGet(e);
  } else if (action === "updateUser") {
    return handleUpdateUserGet(e);
  } else if (action === "deleteUser") {
    return handleDeleteUserGet(e);
  } else if (action === "createRole") {
    return handleCreateRoleGet(e);
  } else if (action === "updateRole") {
    return handleUpdateRoleGet(e);
  } else if (action === "deleteRole") {
    return handleDeleteRoleGet(e);
  } else if (action === "getRoleUsage") {
    return handleGetRoleUsageGet(e);
  } else if (action === "getAllMenus") {
    return handleGetAllMenusGet(e);
  } else if (action === "getRoleMenuAccess") {
    return handleGetRoleMenuAccessGet(e);
  } else if (action === "updateRoleMenuAccess") {
    return handleUpdateRoleMenuAccessGet(e);
  } else if (action === "createMenu") {
    return handleCreateMenuGet(e);
  } else if (action === "updateMenu") {
    return handleUpdateMenuGet(e);
  } else if (action === "deleteMenu") {
    return handleDeleteMenuGet(e);
  } else {
    var errorResult = { status: "error", message: "Invalid action" };
    var callback = e.parameter.callback;
    if (callback) {
      return ContentService.createTextOutput(callback + "(" + JSON.stringify(errorResult) + ")")
        .setMimeType(ContentService.MimeType.JAVASCRIPT);
    }
    return createJsonResponse(errorResult);
  }
}

// ============================================
// LOGIN HANDLER
// ============================================

function handleLogin(requestData, e) {
  var loginid = requestData.loginid;
  var password = requestData.password;
  var ipAddress = getClientIP(e);
  
  // Validate input
  if (!loginid || !password) {
    logLoginAttempt(ipAddress, loginid, false, "Missing credentials");
    return createJsonResponse({ 
      status: "error", 
      message: "Missing credentials" 
    });
  }
  
  var ss = SpreadsheetApp.openById(SHEET_ID);
  
  // Check rate limiting
  var rateLimitResult = checkRateLimit(ss, ipAddress, loginid);
  if (!rateLimitResult.allowed) {
    logLoginAttempt(ipAddress, loginid, false, rateLimitResult.message);
    return createJsonResponse({ 
      status: "error", 
      message: rateLimitResult.message
    });
  }
  
  // Check credentials
  var loginSheet = ss.getSheetByName("LogIn");
  var loginData = loginSheet.getDataRange().getValues();
  
  for (var i = 1; i < loginData.length; i++) {
    if (loginData[i][0] == loginid && loginData[i][2] == password) {
      var roleId = loginData[i][3];
      var userName = loginData[i][1];
      
      // Get role name
      var rolesSheet = ss.getSheetByName("Roles");
      var rolesData = rolesSheet.getDataRange().getValues();
      var roleName = "";
      
      for (var j = 1; j < rolesData.length; j++) {
        if (rolesData[j][0] == roleId) {
          roleName = rolesData[j][1];
          break;
        }
      }
      
      // Get menu access
      var menuAccess = getRoleMenuAccess(ss, roleId);
      
      // Create session
      var sessionId = generateSessionId();
      var now = new Date().getTime();
      var expiryTime = now + SESSION_DURATION;
      
      // Store session in ActiveSessions tab
      var sessionsSheet = ss.getSheetByName("ActiveSessions");
      sessionsSheet.appendRow([
        sessionId,
        loginid,
        userName,
        roleId,
        new Date(now),
        new Date(expiryTime),
        ipAddress,
        new Date(now)
      ]);
      
      // Log successful login
      logLoginAttempt(ipAddress, loginid, true, "Login successful");
      
      // Return session data
      return createJsonResponse({ 
        status: "success", 
        sessionId: sessionId,
        userId: loginid,
        userName: userName,
        roleId: roleId,
        roleName: roleName,
        menuAccess: menuAccess,
        expiryTime: expiryTime
      });
    }
  }
  
  // Invalid credentials
  logLoginAttempt(ipAddress, loginid, false, "Invalid credentials");
  return createJsonResponse({ 
    status: "error", 
    message: "Invalid credentials" 
  });
}

// Validate session
function validateSession(ss, sessionId, userId, ipAddress) {
  var sessionsSheet = ss.getSheetByName("ActiveSessions");
  var data = sessionsSheet.getDataRange().getValues();
  var now = new Date().getTime();
  
  for (var i = 1; i < data.length; i++) {
    if (data[i][0] == sessionId && data[i][1] == userId) {
      var expiryTime = new Date(data[i][5]).getTime();
      var sessionIP = data[i][6];
      
      // Check if expired
      if (now > expiryTime) {
        sessionsSheet.deleteRow(i + 1);
        return { 
          status: "error",
          valid: false, 
          message: "Session expired" 
        };
      }
      
      // Check IP address match
      if (sessionIP !== ipAddress) {
        return { 
          status: "error",
          valid: false, 
          message: "IP address mismatch - security violation" 
        };
      }
      
      // Update last activity
      sessionsSheet.getRange(i + 1, 8).setValue(new Date());
      
      // Get user data
      var userName = data[i][2];
      var roleId = data[i][3];
      
      // Get role name
      var rolesSheet = ss.getSheetByName("Roles");
      var rolesData = rolesSheet.getDataRange().getValues();
      var roleName = "";
      
      for (var j = 1; j < rolesData.length; j++) {
        if (rolesData[j][0] == roleId) {
          roleName = rolesData[j][1];
          break;
        }
      }
      
      // Get menu access
      var menuAccess = getRoleMenuAccess(ss, roleId);
      
      return { 
        status: "success",
        valid: true,
        userId: userId,
        userName: userName,
        roleId: roleId,
        roleName: roleName,
        menuAccess: menuAccess,
        expiryTime: expiryTime
      };
    }
  }
  
  return { 
    status: "error",
    valid: false, 
    message: "Invalid session" 
  };
}

// Check rate limiting
function checkRateLimit(ss, ipAddress, userId) {
  var attemptsSheet = ss.getSheetByName("LoginAttempts");
  var data = attemptsSheet.getDataRange().getValues();
  var now = new Date().getTime();
  var windowStart = now - ATTEMPT_WINDOW;
  
  var ipAttempts = 0;
  var userAttempts = 0;
  
  for (var i = 1; i < data.length; i++) {
    var attemptTime = new Date(data[i][0]).getTime();
    if (attemptTime > windowStart) {
      var attemptIP = data[i][1];
      var attemptUser = data[i][2];
      var success = data[i][3];
      
      if (!success) {
        if (attemptIP === ipAddress) {
          ipAttempts++;
        }
        if (attemptUser === userId) {
          userAttempts++;
        }
      }
    }
  }
  
  if (ipAttempts >= MAX_LOGIN_ATTEMPTS) {
    return { 
      allowed: false, 
      message: "Too many failed attempts from this IP. Try again in 15 minutes." 
    };
  }
  
  if (userAttempts >= MAX_LOGIN_ATTEMPTS) {
    return { 
      allowed: false, 
      message: "Too many failed attempts for this user. Try again in 15 minutes." 
    };
  }
  
  return { allowed: true };
}

// Log login attempt
function logLoginAttempt(ipAddress, userId, success, message) {
  try {
    var ss = SpreadsheetApp.openById(SHEET_ID);
    var attemptsSheet = ss.getSheetByName("LoginAttempts");
    attemptsSheet.appendRow([
      new Date(),
      ipAddress,
      userId,
      success,
      message
    ]);
  } catch (error) {
    // Fail silently if logging fails
  }
}

// Generate secure session ID
function generateSessionId() {
  return 'sess_' + Utilities.getUuid();
}

// Get client IP address
function getClientIP(e) {
  try {
    // Try to get real IP from headers
    if (e.parameter && e.parameter['X-Forwarded-For']) {
      return e.parameter['X-Forwarded-For'].split(',')[0].trim();
    }
    // Fallback to a placeholder
    return 'unknown';
  } catch (error) {
    return 'unknown';
  }
}

// Get menu access for role
function getRoleMenuAccess(ss, roleId) {
  var roleMenuSheet = ss.getSheetByName("RoleMenuAccess");
  var roleMenuData = roleMenuSheet.getDataRange().getValues();
  
  var menuMasterSheet = ss.getSheetByName("MenuMaster");
  var menuMasterData = menuMasterSheet.getDataRange().getValues();
  
  var allowedMenuIds = [];
  for (var i = 1; i < roleMenuData.length; i++) {
    if (roleMenuData[i][0] == roleId) {
      allowedMenuIds.push(roleMenuData[i][1]);
    }
  }
  
  var menus = [];
  for (var j = 1; j < menuMasterData.length; j++) {
    var menuId = menuMasterData[j][0];
    
    if (allowedMenuIds.indexOf(menuId) !== -1) {
      menus.push({
        menuId: menuId,
        parentMenuId: menuMasterData[j][1],
        menuType: menuMasterData[j][2],
        menuName: menuMasterData[j][3],
        pageUrl: menuMasterData[j][4],
        sortOrder: menuMasterData[j][5]
      });
    }
  }
  
  menus.sort(function(a, b) {
    return a.sortOrder - b.sortOrder;
  });
  
  return menus;
}

// Get all users
function handleGetUsersGet(e) {
  var sessionId = e.parameter.sessionId;
  var userId = e.parameter.userId;
  var callback = e.parameter.callback;
  var ipAddress = getClientIP(e);
  
  // Validate session
  var ss = SpreadsheetApp.openById(SHEET_ID);
  var validation = validateSession(ss, sessionId, userId, ipAddress);
  
  if (!validation.valid) {
    var errorResult = { status: "error", message: "Invalid or expired session" };
    if (callback) {
      return ContentService.createTextOutput(callback + "(" + JSON.stringify(errorResult) + ")")
        .setMimeType(ContentService.MimeType.JAVASCRIPT);
    }
    return createJsonResponse(errorResult);
  }
  
  // Check if user is admin
  if (validation.roleName !== "Admin") {
    var errorResult = { status: "error", message: "Access denied: Admin role required" };
    if (callback) {
      return ContentService.createTextOutput(callback + "(" + JSON.stringify(errorResult) + ")")
        .setMimeType(ContentService.MimeType.JAVASCRIPT);
    }
    return createJsonResponse(errorResult);
  }
  
  // Get all users from LogIn sheet
  var loginSheet = ss.getSheetByName("LogIn");
  var loginData = loginSheet.getDataRange().getValues();
  var users = [];
  
  for (var i = 1; i < loginData.length; i++) {
    users.push({
      loginId: loginData[i][0],
      userName: loginData[i][1],
      password: loginData[i][2],
      roleId: loginData[i][3]
    });
  }
  
  var result = { status: "success", users: users };
  if (callback) {
    return ContentService.createTextOutput(callback + "(" + JSON.stringify(result) + ")")
      .setMimeType(ContentService.MimeType.JAVASCRIPT);
  }
  return createJsonResponse(result);
}

  var menuMasterSheet = ss.getSheetByName("MenuMaster");
  var menuMasterData = menuMasterSheet.getDataRange().getValues();
  
  // Find and update menu
  var found = false;
  for (var i = 1; i < menuMasterData.length; i++) {
    if (menuMasterData[i][0] == menuId) {
      menuMasterSheet.getRange(i + 1, 2).setValue(parentMenuId === 'NULL' ? '' : parentMenuId);
      menuMasterSheet.getRange(i + 1, 3).setValue(menuType);
      menuMasterSheet.getRange(i + 1, 4).setValue(menuName);
      menuMasterSheet.getRange(i + 1, 5).setValue(pageUrl === 'NULL' ? '' : pageUrl);
      menuMasterSheet.getRange(i + 1, 6).setValue(sortOrder);
      found = true;
      break;
    }
  }
  
  if (!found) {
    var errorResult = { status: "error", message: "Menu not found" };
    if (callback) {
      return ContentService.createTextOutput(callback + "(" + JSON.stringify(errorResult) + ")")
        .setMimeType(ContentService.MimeType.JAVASCRIPT);
    }
    return createJsonResponse(errorResult);
  }
  
  var result = { status: "success", message: "Menu updated successfully" };
  if (callback) {
    return ContentService.createTextOutput(callback + "(" + JSON.stringify(result) + ")")
      .setMimeType(ContentService.MimeType.JAVASCRIPT);
  }
  return createJsonResponse(result);
}

// Delete menu
function handleDeleteMenuGet(e) {
  var sessionId = e.parameter.sessionId;
  var userId = e.parameter.userId;
  var menuId = e.parameter.menuId;
  var callback = e.parameter.callback;
  var ipAddress = getClientIP(e);
  
  // Validate session
  var ss = SpreadsheetApp.openById(SHEET_ID);
  var validation = validateSession(ss, sessionId, userId, ipAddress);
  
  if (!validation.valid) {
    var errorResult = { status: "error", message: "Invalid or expired session" };
    if (callback) {
      return ContentService.createTextOutput(callback + "(" + JSON.stringify(errorResult) + ")")
        .setMimeType(ContentService.MimeType.JAVASCRIPT);
    }
    return createJsonResponse(errorResult);
  }
  
  // Check if user is admin
  if (validation.roleName !== "Admin") {
    var errorResult = { status: "error", message: "Access denied: Admin role required" };
    if (callback) {
      return ContentService.createTextOutput(callback + "(" + JSON.stringify(errorResult) + ")")
        .setMimeType(ContentService.MimeType.JAVASCRIPT);
    }
    return createJsonResponse(errorResult);
  }
  
  var menuMasterSheet = ss.getSheetByName("MenuMaster");
  var menuMasterData = menuMasterSheet.getDataRange().getValues();
  
  // Find and delete menu
  var found = false;
  for (var i = 1; i < menuMasterData.length; i++) {
    if (menuMasterData[i][0] == menuId) {
      menuMasterSheet.deleteRow(i + 1);
      found = true;
      break;
    }
  }
  
  if (!found) {
    var errorResult = { status: "error", message: "Menu not found" };
    if (callback) {
      return ContentService.createTextOutput(callback + "(" + JSON.stringify(errorResult) + ")")
        .setMimeType(ContentService.MimeType.JAVASCRIPT);
    }
    return createJsonResponse(errorResult);
  }
  
  var result = { status: "success", message: "Menu deleted successfully" };
  if (callback) {
    return ContentService.createTextOutput(callback + "(" + JSON.stringify(result) + ")")
      .setMimeType(ContentService.MimeType.JAVASCRIPT);
  }
  return createJsonResponse(result);
}

// CLEANUP FUNCTION (Run periodically)
// ============================================

function cleanupExpiredSessions() {
  var ss = SpreadsheetApp.openById(SHEET_ID);
  var sessionsSheet = ss.getSheetByName("ActiveSessions");
  var data = sessionsSheet.getDataRange().getValues();
  var now = new Date().getTime();
  
  // Delete from bottom to top to avoid index issues
  for (var i = data.length - 1; i >= 1; i--) {
    var expiryTime = new Date(data[i][5]).getTime();
    if (now > expiryTime) {
      sessionsSheet.deleteRow(i + 1);
    }
  }
  
  Logger.log("Cleaned up expired sessions");
}
